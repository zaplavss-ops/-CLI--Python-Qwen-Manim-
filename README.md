### –õ–æ–≥–∏–∫–∞ Self-Correction

–≠—Ç–æ —Ñ–∏—á–∞ –Ω–∞—à–µ–≥–æ —Å–∫—Ä–∏–ø—Ç–∞. –ú—ã –Ω–µ –ø—Ä–æ—Å—Ç–æ –∑–∞–ø—É—Å–∫–∞–µ–º –∫–æ–¥, –º—ã —Å–ª–µ–¥–∏–º –∑–∞ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–º.

1. **–ì–µ–Ω–µ—Ä–∞—Ü–∏—è:** –ü–æ–ª—É—á–∞–µ–º –∫–æ–¥ –æ—Ç Qwen.
2. **–†–µ–Ω–¥–µ—Ä:** –ó–∞–ø—É—Å–∫–∞–µ–º `manim -qm scene.py`.
3. **–ü—Ä–æ–≤–µ—Ä–∫–∞:** –°–º–æ—Ç—Ä–∏–º –Ω–∞ `returncode`.
   * `0` ‚Äî –£—Å–ø–µ—Ö.

4. **–ò—Å–ø—Ä–∞–≤–ª–µ–Ω–∏—è:** –ï—Å–ª–∏ –æ—à–∏–±–∫–∞ –ø—Ä–æ–∏–∑–æ—à–ª–∞, –º—ã –±–µ—Ä–µ–º —Ç–µ–∫—Å—Ç –æ—à–∏–±–∫–∏ (Python Traceback), —Å–∫–ª–µ–∏–≤–∞–µ–º –µ–≥–æ —Å –Ω–µ—Ä–∞–±–æ—Ç–∞—é—â–∏–º –∫–æ–¥–æ–º –∏ –æ—Ç–ø—Ä–∞–≤–ª—è–µ–º –æ–±—Ä–∞—Ç–Ω–æ –≤ Qwen —Å –ø—Ä–∏–ø–∏—Å–∫–æ–π: *"Fix this error"*.

### –ò—Ç–æ–≥–æ–≤—ã–π –∫–æ–¥

–í–æ—Ç –ø–æ–ª–Ω—ã–π —Å–∫—Ä–∏–ø—Ç, –∫–æ—Ç–æ—Ä—ã–π —Å–æ–µ–¥–∏–Ω—è–µ—Ç –≤—Å—ë –≤–æ–µ–¥–∏–Ω–æ. –û–Ω –Ω–µ —Ç—Ä–µ–±—É–µ—Ç —É—Å—Ç–∞–Ω–æ–≤–∫–∏ —Ç—è–∂–µ–ª—ã—Ö –±–∏–±–ª–∏–æ—Ç–µ–∫ –¥–ª—è LLM, —Ç–æ–ª—å–∫–æ —Å—Ç–∞–Ω–¥–∞—Ä—Ç–Ω—ã–µ `subprocess`, `re` –∏ `sys`.

```python

import sys
import re
import os

# --- –ö–û–ù–§–ò–ì–£–†–ê–¶–ò–Ø ---
MAX_RETRIES = 3  # –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø–æ–ø—ã—Ç–æ–∫ –∞–≤—Ç–æ-–∏—Å–ø—Ä–∞–≤–ª–µ–Ω–∏—è
OUTPUT_FILE = "scene.py"
SCENE_NAME = "GenScene"  # –î–æ–ª–∂–Ω–æ —Å–æ–≤–ø–∞–¥–∞—Ç—å —Å –∏–º–µ–Ω–µ–º –∫–ª–∞—Å—Å–∞ –≤ System Prompt

# System Prompt –∑–∞–¥–∞–µ—Ç —Ä–æ–ª—å –∏ –æ–≥—Ä–∞–Ω–∏—á–µ–Ω–∏—è
SYSTEM_PROMPT = """

Your task is to write a complete, runnable Python script.

RULES:
1. Output ONLY the Python code. No markdown, no explanations.
2. Start with `from manim import *`.
3. Create a class named `GenScene` inheriting from `Scene`.
4. Use `def construct(self):` method.

def strip_ansi(text):
    """
    –ß–∏—Å—Ç–∏–º –≤—ã–≤–æ–¥ –æ—Ç —Ü–≤–µ—Ç–æ–≤ –∏ —É–ø—Ä–∞–≤–ª—è—é—â–∏—Ö —Å–∏–º–≤–æ–ª–æ–≤ –∫–æ–Ω—Å–æ–ª–∏,
    –∫–æ—Ç–æ—Ä—ã–µ –≥–µ–Ω–µ—Ä–∏—Ä—É–µ—Ç –∏–Ω—Ç–µ—Ä–∞–∫—Ç–∏–≤–Ω—ã–π CLI.
    """
    ansi_escape = re.compile(r'\x1B(?:[@-Z\\-_]|\[[0-?]*[ -/]*[@-~])')
    return ansi_escape.sub('', text)

def call_qwen(prompt_text):
    """
    –í—ã–∑—ã–≤–∞–µ—Ç qwen-cli –≤ '–Ω–µ–º–æ–º' —Ä–µ–∂–∏–º–µ.

    input_payload = f"{full_prompt}\n/quit\n"

        result = subprocess.run(
            [command], 
            input=input_payload,
            capture_output=True,
            text=True,
            encoding='utf-8', # Node.js –æ–±—ã—á–Ω–æ –æ—Ç–¥–∞–µ—Ç UTF-8
            check=False
        )
        # –í–æ–∑–≤—Ä–∞—â–∞–µ–º —á–∏—Å—Ç—ã–π —Ç–µ–∫—Å—Ç –±–µ–∑ –º—É—Å–æ—Ä–∞
        return strip_ansi(result.stdout)
        
    except FileNotFoundError:
        print("‚ùå –û—à–∏–±–∫–∞: –ö–æ–º–∞–Ω–¥–∞ 'qwen' –Ω–µ –Ω–∞–π–¥–µ–Ω–∞. –£—Å—Ç–∞–Ω–æ–≤–∏—Ç–µ: npm install -g @qwen-code/qwen-code")
        sys.exit(1)

def extract_code(raw_text):
    """–í—ã—Ç–∞—Å–∫–∏–≤–∞–µ—Ç Python-–∫–æ–¥ –∏–∑ –æ—Ç–≤–µ—Ç–∞ –º–æ–¥–µ–ª–∏."""
    if not raw_text: return ""
    
    # –ò—â–µ–º –±–ª–æ–∫ –∫–æ–¥–∞ –≤ –º–∞—Ä–∫–¥–∞—É–Ω–µ
    match = re.search(r'```python\s*(.*?)```', raw_text, re.DOTALL)
    if match: return match.group(1)
    
    # Fallback: –µ—Å–ª–∏ –º–æ–¥–µ–ª—å –≤–µ—Ä–Ω—É–ª–∞ –∫–æ–¥ –±–µ–∑ –º–∞—Ä–∫–¥–∞—É–Ω–∞
    if "from manim import" in raw_text:
        start = raw_text.find("from manim import")
        return raw_text[start:]
        
    return raw_text

    """–ó–∞–ø—É—Å–∫–∞–µ—Ç —Ä–µ–Ω–¥–µ—Ä –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å—Ç–∞—Ç—É—Å —É—Å–ø–µ—Ö–∞ + –ª–æ–≥."""

    cmd = [sys.executable, "-m", "manim", "-qm", OUTPUT_FILE, SCENE_NAME]

    startupinfo = subprocess.STARTUPINFO() if sys.platform == "win32" else None
    
    result = subprocess.run(
        cmd, capture_output=True, text=True, encoding='utf-8', startupinfo=startupinfo
    )
    return (result.returncode == 0, result.stderr)

def main():
    if len(sys.argv) < 2:
        print(f"Usage: python {sys.argv[0]} \"–û–ø–∏—Å–∞–Ω–∏–µ —Å—Ü–µ–Ω—ã\"")
        return

    print(f"üé¨ –ì–µ–Ω–µ—Ä–∞—Ü–∏—è —Å—Ü–µ–Ω—ã: '{user_topic}'")
```
